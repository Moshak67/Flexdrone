 package com.fdmgroup.flexdronepodq42022.Model;
 
 import jakarta.persistence.*;
 import lombok.AllArgsConstructor;
 import lombok.Getter;
 import lombok.NoArgsConstructor;
 import lombok.Setter;
 
 @Setter
 @Getter
 @NoArgsConstructor
 @AllArgsConstructor
 @Entity
 @Table(name = "products")
 public class Product {
 
     @Id
     @GeneratedValue(strategy = GenerationType.IDENTITY)
     private long id;
 
     @Column(nullable = false, unique = true)
     private String sku;
 
     @Column
     private String description;
 
     @Column
     private float retailPrice;
 
     @Column
     public String category;
 
     @Column
     private String name;
 
     @Column
     private long externalStock;
 
     @Column
     private long internalStock;
 
     @Column
     private long minStockLevel;
 
     @Column
     private boolean isPart;
 
     @Column
     private String externalNote;
 
     @Column
     private String internalNote;
 
     @Column
     private String imageUrl = "dronesAreUs.com.au/orangeDrone.jpg";
 
     private int quantityInCart = 1;
 
     public Product(String sku, String description, float retailPrice, String category, String name, long externalStock,
                    long internalStock, long minStockLevel, boolean isPart, String externalNote, String internalNote, String imageUrl, int quantityInCart) {
 
         super();
         this.sku = sku;
         this.description = description;
         this.retailPrice = retailPrice;
         this.category = category;
         this.name = name;
         this.externalStock = externalStock;
         this.internalStock = internalStock;
         this.minStockLevel = minStockLevel;
         this.isPart = isPart;
         this.externalNote = externalNote;
         this.internalNote = internalNote;
         this.imageUrl = imageUrl;
         this.quantityInCart = quantityInCart;
 
     }
 
     public String toSummaryString() {
         return "Product [retailPrice = " + retailPrice + ", name = " + name + ", quantity = " + quantityInCart + "]";
     }
 
 
 }
